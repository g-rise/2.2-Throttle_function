function debounce(func: Function, delay: number = 1000): Function {
    let timeoutId: ReturnType<typeof setTimeout>

    return function (this: any, ...args: any[]) {
        clearTimeout(timeoutId)

        const contexte = this
        const parametres = args

        timeoutId = setTimeout(() => {
            const resultat = func.apply(contexte, parametres)
            return resultat
        }, delay)
    }

}

function saludar(nom: string, cognom: string): void {
    console.log(`Bon dia ${nom} ${cognom}`)
}

const salut = debounce(saludar)

salut("Magic", "Johnson")
salut("Magic", "Johnson")
salut("Magic", "Johnson")
